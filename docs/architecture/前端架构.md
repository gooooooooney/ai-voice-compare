```
src/

└── README.md           # 部署说明
```

### 组件模板
```typescript
class VoiceComparison {
  constructor() {
    this.deepgramConnected = false;
    this.assemblyaiConnected = false;
    this.isRecording = false;
  }
  
  async init() {
    await this.connectAPIs();
    this.setupEventListeners();
  }
}
```
